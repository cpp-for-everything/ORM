94 constexpr std::size_t type_to_id_extension_apply(std::size_t ext) noexcept {
95  constexpr std::size_t native_id = (Unptr & native_types_mask);
96  constexpr std::size_t extensions = (Unptr & ~native_types_mask);
97  static_assert(

99  "====================> Boost.PFR: Too many extensions for a single field (something close to `int************************** p;` is in the POD type)."

247 constexpr size_array<sizeof(Type) * 3> type_to_id(identity<Type>, std::enable_if_t<!std::is_enum<Type>::value && !std::is_empty<Type>::value && !std::is_union<Type>::value>*) noexcept {

249  size_array<sizeof(Type) * 3> result {{tuple_begin_tag}};
250  constexpr bool requires_tuplening = (
251  (t.count_nonzeros() != 1) || (t.count_nonzeros() == t.count_from_opening_till_matching_parenthis_seq(0, tuple_begin_tag, tuple_end_tag))
252  );

429  for (std::size_t i = 0; i < subtuple_lengths.size(); ++i) {
430  if (subtuple_lengths.data[i]) {

471  size_t_< a.data[ First + type_indexes.data[INew] ] >{}, // id of type
472  size_t_< First + type_indexes.data[INew] >{}, // index of current id in `a`
