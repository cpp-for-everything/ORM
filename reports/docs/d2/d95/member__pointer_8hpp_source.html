80  if constexpr (op1.operator std::string_view() != op2.operator std::string_view()) {
81  return false;
82  }
83  // Different operand types
84  else if constexpr (requires { {Statement1._1.operation};} ^ requires { {Statement2._1.operation};}) {
85  return false;
86  }
87  else if constexpr (requires { {Statement1._2.operation};} ^ requires { {Statement2._2.operation};}) {
88  return false;
89  }
90  else {
91  bool answer = true;

93  if constexpr (requires { {Statement1._1.operation};} && requires { {Statement2._1.operation};}) {
