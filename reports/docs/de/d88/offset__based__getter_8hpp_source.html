25  // Our own implementation of std::aligned_storage. On godbolt with MSVC, I have compilation errors
26  // using the standard version, it seems the compiler cannot generate default ctor.

28  template <std::size_t s, std::size_t a> struct internal_aligned_storage
29  {
30  alignas(a) char storage_[s];
31  };

33  // Metafunction that replaces tuple<T1, T2, T3, ...> with
34  // tuple<std::aligned_storage_t<sizeof(T1), alignof(T1)>, std::aligned_storage<sizeof(T2), alignof(T2)>, ...>
35  //
36  // The point is, the new tuple is "layout compatible" in the sense that members have the same offsets,
37  // but this tuple is constexpr constructible.
38 
39  template <typename T> struct tuple_of_aligned_storage;
40 
41  template <typename... Ts> struct tuple_of_aligned_storage<sequence_tuple::tuple<Ts...>>
42  {
43  using type = sequence_tuple::tuple<internal_aligned_storage<
44  sizeof(Ts),
45 #if defined(__GNUC__) && __GNUC__ < 8 && !defined(__x86_64__) && !defined(__CYGWIN__)
46  // Before GCC-8 the `alignof` was returning the optimal alignment rather than the minimal one.
47  // We have to adjust the alignemnt because otherwise we get the wrong offset.
48  (alignof(Ts) > 4 ? 4 : alignof(Ts))
49 #else
50  alignof(Ts)
51 #endif
52  >...>;
53  };

59  /***
60  * Given a structure type and its sequence of members, we want to build a function
61  * object "getter" that implements a version of `std::get` using offset arithmetic
62  * and reinterpret_cast.
63  *
64  * typename U should be a user-defined struct
65  * typename S should be a sequence_tuple which is layout compatible with U
66  */
67 
68  template <typename U, typename S> class offset_based_getter
69  {
70  using this_t = offset_based_getter<U, S>;
71

75  "SimpleAggregate?");

91  // Get type of idx'th member
92  template <std::size_t idx> using index_t = typename sequence_tuple::tuple_element<idx, S>::type;
93 
94  // Get offset of idx'th member
95  // Idea: Layout object has the same offsets as instance of S, so if S and U are layout compatible, then these offset
96  // calculations are correct.
97  template <std::size_t idx> static constexpr std::ptrdiff_t offset() noexcept
98  {
99  constexpr tuple_of_aligned_storage_t<S> layout{};
100  return &sequence_tuple::get<idx>(layout).storage_[0] - &sequence_tuple::get<0>(layout).storage_[0];
101  }
