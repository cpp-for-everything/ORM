8 // The Great Type Loophole (C++14)
9 // Initial implementation by Alexandr Poltavsky, http://alexpolt.github.io
10 //
11 // Description:
12 // The Great Type Loophole is a technique that allows to exchange type information with template
13 // instantiations. Basically you can assign and read type information during compile time.
14 // Here it is used to detect data members of a data type. I described it for the first time in
15 // this blog post http://alexpolt.github.io/type-loophole.html .
16 //
17 // This technique exploits the http://www.open-std.org/jtc1/sc22/wg21/docs/cwg_active.html#2118
18 // CWG 2118. Stateful metaprogramming via friend injection
19 // Note: CWG agreed that such techniques should be ill-formed, although the mechanism for prohibiting them is as yet undetermined.

54 // tag<T,N> generates friend declarations and helps with overload resolution.
55 // There are two types: one with the auto return type, which is the way we read types later.
56 // The second one is used in the detection of instantiations without which we'd get multiple
57 // definitions.

59 template <class T, std::size_t N>
60 struct tag {
61  friend auto loophole(tag<T,N>);
62 };

64 // The definitions of friend functions.
65 template <class T, class U, std::size_t N, bool B>
66 struct fn_def_lref {
67  friend auto loophole(tag<T,N>) {
68  // Standard Library containers do not SFINAE on invalid copy constructor. Because of that std::vector<std::unique_ptr<int>> reports that it is copyable,
69  // which leads to an instantiation error at this place.
70  //
71  // To workaround the issue, we check that the type U is movable, and move it in that case.
72  using no_extents_t = std::remove_all_extents_t<U>;
73  return static_cast< std::conditional_t<std::is_move_constructible<no_extents_t>::value, no_extents_t&&, no_extents_t&> >(
74  pfr::detail::unsafe_declval<no_extents_t&>()
75  );
76  }
77 };
78 template <class T, class U, std::size_t N, bool B>
79 struct fn_def_rref {
80  friend auto loophole(tag<T,N>) { return std::move(pfr::detail::unsafe_declval< std::remove_all_extents_t<U>& >()); }
81 };

92 // This has a templated conversion operator which in turn triggers instantiations.
93 // Important point, using sizeof seems to be more reliable. Also default template
94 // arguments are "cached" (I think). To fix that I provide a U template parameter to
95 // the ins functions which do the detection using constexpr friend functions and SFINAE.
96 template <class T, std::size_t N>
97 struct loophole_ubiq_lref {
98  template<class U, std::size_t M> static std::size_t ins(...);
99  template<class U, std::size_t M, std::size_t = sizeof(loophole(tag<T,M>{})) > static char ins(int);
100 
101  template<class U, std::size_t = sizeof(fn_def_lref<T, U, N, sizeof(ins<U, N>(0)) == sizeof(char)>)>
102  constexpr operator U&() const&& noexcept; // `const&&` here helps to avoid ambiguity in loophole instantiations. optional_like test validate that behavior.
103 };
104 
105 template <class T, std::size_t N>
106 struct loophole_ubiq_rref {
107  template<class U, std::size_t M> static std::size_t ins(...);
108  template<class U, std::size_t M, std::size_t = sizeof(loophole(tag<T,M>{})) > static char ins(int);
109 
110  template<class U, std::size_t = sizeof(fn_def_rref<T, U, N, sizeof(ins<U, N>(0)) == sizeof(char)>)>
111  constexpr operator U&&() const&& noexcept; // `const&&` here helps to avoid ambiguity in loophole instantiations. optional_like test validate that behavior.
112 };

151 template <class T>
152 auto tie_as_tuple_loophole_impl(T& lvalue) noexcept {
153  using type = std::remove_cv_t<std::remove_reference_t<T>>;
154  using indexes = detail::make_index_sequence<fields_count<type>()>;
155  using loophole_type_list = typename detail::loophole_type_list_selector<
156  std::is_copy_constructible<std::remove_all_extents_t<type>>::value, type, indexes
157  >::type;
158  using tuple_type = typename loophole_type_list::type;
159 
160  return pfr::detail::make_flat_tuple_of_references(
161  lvalue,
162  offset_based_getter<type, tuple_type>{},
163  size_t_<0>{},
164  size_t_<tuple_type::size_v>{}
165  );
166 }
