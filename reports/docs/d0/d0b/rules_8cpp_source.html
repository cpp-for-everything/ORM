139  // Different operations
140  if constexpr (op1.operator std::string_view() != op2.operator std::string_view()) {
141  return false;
142  }
143  // Different operand types
144  else if constexpr (requires { {rule1._1.operation};} ^ requires { {rule2._1.operation};}) {
145  return false;
146  }
147  else if constexpr (requires { {rule1._2.operation};} ^ requires { {rule2._2.operation};}) {
148  return false;
149  }
150  else {
151  bool answer = true;
152  // Both rules
153  if constexpr (requires { {rule1._1.operation};} && requires { {rule2._1.operation};}) {
154  answer = answer && (are_same(rule1._1, rule2._1));
155  }
156  // Both not rules
157  if constexpr (!requires { {rule1._1.operation};} && requires { {rule2._1.operation};}) {
158  answer = answer && (rule1._1 == rule2._1);
159  }
160  // Both rules

164  // Both not rules
